      const current = window.scrollY;
      if (current > 80) {
        navbar.classList.add('bg-white/90', 'dark:bg-dark-900/90', 'backdrop-blur-md', 'shadow-sm');
      } else {
        navbar.classList.remove('bg-white/90', 'dark:bg-dark-900/90', 'backdrop-blur-md', 'shadow-sm');
      }
      lastScroll = current;

      // Scroll to top button
      const btn = document.getElementById('scrollTopBtn');
      if (current > 500) {
        btn.classList.remove('opacity-0', 'translate-y-4', 'pointer-events-none');
        btn.classList.add('opacity-100', 'translate-y-0', 'pointer-events-auto');
      } else {

    });

    const revealObserver = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.classList.add('active');
        }
      });
    }, { threshold: 0.1, rootMargin: '0px 0px -40px 0px' });

    document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));

    // ====== BREED SEARCH ======
    function searchBreeds() {
      const query = document.getElementById('breedSearch').value.toLowerCase().trim();
      const cards = document.querySelectorAll('.breed-card');
      let visibleCount = 0;

      cards.forEach(card => {
        const breed = card.getAttribute('data-breed');
        const name = card.querySelector('h3').textContent.toLowerCase();
        const desc = card.querySelector('p').textContent.toLowerCase();
        const match = breed.includes(query) || name.includes(query) || desc.includes(query);
        card.style.display = match ? '' : 'none';
        if (match) visibleCount++;
      });

      document.getElementById('noResults').classList.toggle('hidden', visibleCount > 0);
    }

    // ====== LIGHTBOX ======

      const img = el.querySelector('img');
      const lb = document.getElementById('lightbox');
      document.getElementById('lightbox-img').src = img.src;
      lb.classList.add('active');
      document.body.style.overflow = 'hidden';
    }
    function closeLightbox() {
      document.getElementById('lightbox').classList.remove('active');

    // ====== SLIDER ======

    }
    function getMaxIndex() {
      const total = document.querySelectorAll('#sliderTrack > div').length;
      const visible = window.innerWidth >= 1024 ? 3 : window.innerWidth >= 768 ? 2 : 1;
      return Math.max(0, total - visible);
    }
    function updateSlider() {
      const track = document.getElementById('sliderTrack');
      track.style.transform = `translateX(-${sliderIndex * getSlideWidth()}%)`;
    }
    function slideNext() {
      sliderIndex = Math.min(sliderIndex + 1, getMaxIndex());
      updateSlider();
    }
    function slidePrev() {
      sliderIndex = Math.max(sliderIndex - 1, 0);
      updateSlider();
    }
